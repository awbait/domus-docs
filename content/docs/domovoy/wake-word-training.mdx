# Обучение wake word "Домовой"

Пошаговое руководство по обучению кастомной OpenWakeWord модели на русское слово "Домовой".

---

## Что получится на выходе

Файл `domovoy.onnx` (~200KB). Кладём в Docker-контейнер, ставим `WW_MODEL=domovoy` — готово.

---

## Требования

- Google Colab (бесплатный T4 GPU) или Linux-машина с GPU
- SaluteSpeech аккаунт (бесплатно, 200K символов) — или Silero/Piper (уже в проекте)
- Микрофон для записи реальных сэмплов
- 5-10 человек для записи произнесений

---

## Формат аудио (везде одинаковый)

```
WAV, 16 kHz, mono, 16-bit PCM
Длительность клипа: 1-3 секунды
```

Конвертация из любого формата:
```bash
ffmpeg -i input.mp3 -ar 16000 -ac 1 -sample_fmt s16 output.wav
```

## Шаг 2. Генерация синтетических сэмплов (positive)

### Вариант A — через Silero TTS + Piper (бесплатно)

Используем уже имеющиеся TTS-сервисы проекта.

```python
# generate_positive.py
"""Генерация positive сэмплов через Silero TTS."""
import torch
import wave
import struct
import os
import random

# Загрузка Silero
model, _ = torch.hub.load('snakers4/silero-models', model='silero_tts',
                           language='ru', speaker='v5_ru')

VOICES = ['xenia', 'baya', 'kseniya', 'aidar', 'eugene']
PHRASES = [
    'домовой',
    'домовой!',
    'эй домовой',
    'Домовой',
    'домовооой',
]
SAMPLE_RATE = 16000
OUTPUT_DIR = 'positive_clips'
os.makedirs(OUTPUT_DIR, exist_ok=True)

idx = 0
for phrase in PHRASES:
    for voice in VOICES:
        # Вариации скорости (от нормальной до быстрой)
        for speed in [0.8, 0.9, 1.0, 1.1, 1.2]:
            audio = model.apply_tts(
                text=phrase,
                speaker=voice,
                sample_rate=SAMPLE_RATE,
            )
            # Сохранение в WAV 16-bit PCM
            pcm = (audio * 32767).clamp(-32768, 32767).to(torch.int16).numpy()
            path = os.path.join(OUTPUT_DIR, f'pos_{idx:05d}.wav')
            with wave.open(path, 'w') as wf:
                wf.setnchannels(1)
                wf.setsampwidth(2)
                wf.setframerate(SAMPLE_RATE)
                wf.writeframes(pcm.tobytes())
            idx += 1
            print(f'{path}: {phrase} / {voice} / speed={speed}')

print(f'Сгенерировано {idx} positive сэмплов')
```

Результат: ~125 клипов (5 фраз × 5 голосов × 5 скоростей).

### Вариант B — через SaluteSpeech (бесплатно, 200K символов)

6 русских голосов от Сбера. Качество на русском лучше чем OpenAI — нативные русскоязычные голоса.
200K символов = ~25000 произнесений "домовой". Регистрация: https://developers.sber.ru

```python

"""Генерация positive сэмплов через SaluteSpeech REST API."""
# pip install salute-speech
import os
import requests
import uuid
import wave

# Получить токен: https://developers.sber.ru/docs/ru/salutespeech/authentication
AUTH_TOKEN = os.environ['SALUTE_TOKEN']  # Bearer токен
OUTPUT_DIR = 'positive_clips_salute'
os.makedirs(OUTPUT_DIR, exist_ok=True)

VOICES = ['Nec_24000', 'Bys_24000', 'May_24000', 'Tur_24000', 'Ost_24000', 'Pon_24000']
PHRASES = [
    'домовой',
    'домовой!',
    'эй домовой',
    'Домовой',
    'домовооой',
    'эй, домовой!',
    'домовой, слушай',
]

API_URL = 'https://smartspeech.sber.ru/rest/v1/text:synthesize'

idx = 0
for phrase in PHRASES:
    for voice in VOICES:
        for _ in range(20):  # 20 вариаций на комбинацию
            resp = requests.post(
                API_URL,
                headers={
                    'Authorization': f'Bearer {AUTH_TOKEN}',
                    'Content-Type': 'application/text',
                },
                params={'voice': voice, 'format': 'wav16'},
                data=phrase.encode('utf-8'),
            )
            if resp.status_code == 200:
                path = os.path.join(OUTPUT_DIR, f'pos_{idx:05d}.wav')
                with open(path, 'wb') as f:
                    f.write(resp.content)
                idx += 1
            else:
                print(f'Ошибка {resp.status_code}: {resp.text}')

# Конвертировать в 16kHz mono:
# for f in positive_clips_salute/*.wav; do
#   ffmpeg -y -i "$f" -ar 16000 -ac 1 -sample_fmt s16 "${f%.wav}_16k.wav"
# done

print(f'Сгенерировано {idx} positive сэмплов')
```

### Вариант C — через ElevenLabs (бесплатно, 10K credits/мес)

Преимущество: voice cloning. Загрузить 1-5 мин голоса реального человека → клонировать → генерить "домовой" этим голосом. Заменяет этап voice conversion (шаг 4).

```python
# generate_positive_elevenlabs.py
"""Генерация positive сэмплов через ElevenLabs API."""
# pip install elevenlabs
import os
from elevenlabs.client import ElevenLabs

client = ElevenLabs(api_key=os.environ['ELEVENLABS_API_KEY'])
OUTPUT_DIR = 'positive_clips_elevenlabs'
os.makedirs(OUTPUT_DIR, exist_ok=True)

PHRASES = [
    'домовой', 'домовой!', 'эй домовой',
    'Домовой', 'домовооой', 'эй, домовой!',
]

# Встроенные голоса (русский поддерживается)
VOICE_IDS = ['Rachel', 'Clyde', 'Domi', 'Bella', 'Antoni', 'Thomas']

idx = 0
for phrase in PHRASES:
    for voice in VOICE_IDS:
        audio = client.text_to_speech.convert(
            text=phrase,
            voice_id=voice,
            model_id='eleven_multilingual_v2',
            output_format='pcm_16000',
        )
        path = os.path.join(OUTPUT_DIR, f'pos_{idx:05d}.raw')
        with open(path, 'wb') as f:
            for chunk in audio:
                f.write(chunk)
        # Конвертировать PCM → WAV потом через ffmpeg
        idx += 1

# ffmpeg -f s16le -ar 16000 -ac 1 -i pos_00000.raw pos_00000.wav

print(f'Сгенерировано {idx} сэмплов')
```

10K credits ≈ ~1250 произнесений. Меньше чем SaluteSpeech, но:
- Voice cloning: загрузить записи родственников → генерить их голосами без записи 20-30 фраз вручную
- 30+ голосов с поддержкой русского
- Лимит обновляется каждый месяц

## Шаг 5. Запись реальных сэмплов от людей

### Инструкция для записи

**Формат:** WAV, 16 kHz, mono, 16-bit.

**На компьютере** (самый простой способ):
```bash
# Установить sox
# Linux: apt install sox
# macOS: brew install sox

# Записать один клип (3 секунды):
sox -d -r 16000 -c 1 -b 16 sample_001.wav trim 0 3

# Или непрерывно записывать, потом нарезать
sox -d -r 16000 -c 1 -b 16 session_ivan.wav
# Ctrl+C чтобы остановить
```

**На телефоне:**
- Любое приложение диктофона (Voice Recorder, Easy Voice Recorder)
- Записать одним файлом все произнесения подряд с паузами
- Перенести на комп и конвертировать:
  ```bash
  ffmpeg -i phone_recording.m4a -ar 16000 -ac 1 -sample_fmt s16 recording.wav
  ```
- Нарезать на отдельные клипы (вручную в Audacity или скриптом)

### Что записывать

Каждый человек произносит **~40 раз**. Между произнесениями пауза 2-3 секунды (для нарезки).
Записывать одним файлом — потом скрипт нарежет по паузам.

**Блок 1 — Базовое произнесение (10 раз)**
Просто "домовой" нормальным голосом. Сидя перед микрофоном, ~0.5м.
Зачем: основа датасета, самый частый сценарий.

**Блок 2 — Громкость (6 раз)**
- "домовой" громко, как зовёшь из другой комнаты × 3
- "домовой" тихо, почти шёпотом, как ночью × 3

Зачем: модель должна срабатывать и на шёпот, и на крик.

**Блок 3 — Дистанция (6 раз)**
- Встать в 1 метре от микрофона, "домовой" × 3
- Встать в 2-3 метрах, "домовой" × 3

Зачем: в реальности человек не всегда рядом с колонкой.

**Блок 4 — Скорость (4 раза)**
- "домовой" быстро, скороговоркой × 2
- "домовооой" растянуто, лениво × 2

Зачем: все говорят с разной скоростью.

**Блок 5 — С контекстом (6 раз)**
- "эй домовой" × 2
- "домовой, который час?" × 2
- "слушай домовой" × 2

Зачем: в реальности слово не произносится изолированно, перед и после него речь.

**Блок 6 — С фоновым шумом (6 раз)**
Включить ТВ или музыку на средней громкости:
- "домовой" нормально × 3
- "домовой" громко (перекрикивая фон) × 3

Зачем: самый сложный сценарий, именно тут модели ошибаются чаще всего.

**Блок 7 — Эмоции (4 раза)**
- "домовой" сонным голосом, только проснулся × 2
- "домовой!" раздражённо / нетерпеливо × 2

Зачем: голос сильно меняется в зависимости от состояния.

**Итого ~42 произнесения на человека. 5 человек = 210, 10 человек = 420 клипов.**

### Кого записывать

Чем разнообразнее — тем лучше:
- Мужчины и женщины (разный тембр)
- Разный возраст (дети, взрослые, пожилые)
- Разный акцент (если есть)

Минимум: **5 человек**. Хорошо: **8-10**. Идеально: **15+**.

### Памятка для записывающего

Распечатать или скинуть в мессенджер:

```
Запись wake word "домовой" для голосового ассистента.

Формат: одним файлом, с паузами 2-3 сек между словами.
Микрофон: телефон или компьютер, начать в тихой комнате.
```

### Нарезка длинной записи на клипы

Если человек записал всё одним файлом с паузами:

```python
# split_recording.py
"""Нарезка длинной записи на отдельные клипы по тишине."""
import wave
import numpy as np
import os

def split_by_silence(input_path, output_dir, silence_thresh=500, min_duration=0.3):
    """Разрезать WAV по паузам."""
    os.makedirs(output_dir, exist_ok=True)

    with wave.open(input_path, 'rb') as wf:
        sr = wf.getframerate()
        pcm = np.frombuffer(wf.readframes(wf.getnframes()), dtype=np.int16)

    # RMS по чанкам 100ms
    chunk_size = sr // 10
    is_speech = []
    for i in range(0, len(pcm) - chunk_size, chunk_size):
        rms = np.sqrt(np.mean(pcm[i:i+chunk_size].astype(float)**2))
        is_speech.append(rms > silence_thresh)

    # Найти сегменты речи
    segments = []
    in_segment = False
    start = 0
    for i, speech in enumerate(is_speech):
        if speech and not in_segment:
            start = i * chunk_size
            in_segment = True
        elif not speech and in_segment:
            end = i * chunk_size
            if (end - start) / sr >= min_duration:
                # Добавить padding 200ms
                segments.append((max(0, start - sr//5), min(len(pcm), end + sr//5)))
            in_segment = False

    # Сохранить клипы
    for idx, (s, e) in enumerate(segments):
        clip = pcm[s:e]
        path = os.path.join(output_dir, f'clip_{idx:03d}.wav')
        with wave.open(path, 'w') as wf:
            wf.setnchannels(1)
            wf.setsampwidth(2)
            wf.setframerate(sr)
            wf.writeframes(clip.tobytes())
        print(f'{path}: {(e-s)/sr:.1f}s')

    print(f'Нарезано {len(segments)} клипов')

# Использование:
split_by_silence('session_ivan.wav', 'real_recordings/ivan/')
```

## Шаг 7. Аугментация

Все positive сэмплы (синтетика + voice conversion + реальные) обрабатываем:

```python
# augment.py
"""Аугментация positive сэмплов."""
import numpy as np
import wave
import os
import glob
import random
from audiomentations import (
    Compose, AddBackgroundNoise, ApplyImpulseResponse,
    Gain, PitchShift, TimeStretch,
)

augmenter = Compose([
    AddBackgroundNoise(
        sounds_path='home_negatives/',   # фоновый шум квартиры
        min_snr_db=5, max_snr_db=30, p=0.7,
    ),
    ApplyImpulseResponse(
        ir_path='mit_rirs/',             # реверберация помещений
        p=0.5,
    ),
    Gain(min_gain_db=-12, max_gain_db=6, p=0.8),
    PitchShift(min_semitones=-2, max_semitones=2, p=0.3),
    TimeStretch(min_rate=0.85, max_rate=1.15, p=0.3),
])

INPUT_DIRS = ['positive_clips/', 'positive_clips_vc/', 'real_recordings/*/']
OUTPUT_DIR = 'augmented_positive/'
os.makedirs(OUTPUT_DIR, exist_ok=True)
AUGMENTATIONS_PER_CLIP = 3

idx = 0
for pattern in INPUT_DIRS:
    for clip_path in glob.glob(pattern + '*.wav'):
        with wave.open(clip_path, 'rb') as wf:
            sr = wf.getframerate()
            pcm = np.frombuffer(wf.readframes(wf.getnframes()), dtype=np.int16)
        audio_float = pcm.astype(np.float32) / 32768.0

        for _ in range(AUGMENTATIONS_PER_CLIP):
            augmented = augmenter(samples=audio_float, sample_rate=sr)
            out_pcm = (augmented * 32767).clip(-32768, 32767).astype(np.int16)
            path = os.path.join(OUTPUT_DIR, f'aug_{idx:06d}.wav')
            with wave.open(path, 'w') as wf:
                wf.setnchannels(1)
                wf.setsampwidth(2)
                wf.setframerate(sr)
                wf.writeframes(out_pcm.tobytes())
            idx += 1

print(f'Аугментировано: {idx} клипов')
```

```python
# test_model.py
"""Тестирование модели на WAV файлах."""
from openwakeword.model import Model
import numpy as np
import wave
import glob

model = Model(
    wakeword_models=['./domovoy_model/domovoy.onnx'],
    inference_framework='onnx',

# Тест на positive (должно срабатывать)
print('=== POSITIVE (должно быть > 0.5) ===')
hits, total = 0, 0
for f in glob.glob('real_recordings/*/*.wav')[:20]:
    with wave.open(f, 'rb') as wf:
        pcm = np.frombuffer(wf.readframes(wf.getnframes()), dtype=np.int16)
    max_score = 0
    for i in range(0, len(pcm) - 1280, 1280):
        pred = model.predict(pcm[i:i+1280])
        score = list(pred.values())[0]
        max_score = max(max_score, score)
    status = 'HIT' if max_score > 0.5 else 'MISS'
    if max_score > 0.5:
        hits += 1
    total += 1
    print(f'  {status} {max_score:.3f} {f}')
print(f'  Recall: {hits}/{total} = {hits/total*100:.0f}%')

# Тест на negative (не должно срабатывать)
print('\n=== NEGATIVE (должно быть &lt; 0.5) ===')
fps, total = 0, 0
for f in glob.glob('negative_clips/*.wav')[:20]:
    with wave.open(f, 'rb') as wf:
        pcm = np.frombuffer(wf.readframes(wf.getnframes()), dtype=np.int16)
    max_score = 0
    for i in range(0, len(pcm) - 1280, 1280):
        pred = model.predict(pcm[i:i+1280])
        score = list(pred.values())[0]
        max_score = max(max_score, score)
    status = 'FP!' if max_score > 0.5 else 'OK '
    if max_score > 0.5:
        fps += 1
    total += 1
    print(f'  {status} {max_score:.3f} {f}')
print(f'  False positives: {fps}/{total}')
```

### Тюнинг порога

| `WW_THRESHOLD` | Эффект |
|---|---|
| 0.3 | Чувствительный — больше срабатываний, больше ложных |
| 0.5 | Баланс (начать с этого) |
| 0.7 | Строгий — меньше ложных, но может пропускать |

Подбирать по ощущениям: если пропускает — снизить, если ложные — повысить.
